public|escolas_view|false| SELECT e.id,
    e.nome,
    e.status,
    e.plano_atual,
    e.plano_atual::text AS plano,
    (( SELECT max(al.created_at) AS max
           FROM audit_logs al
          WHERE al.escola_id = e.id))::timestamp without time zone AS last_access,
    COALESCE(a.total_alunos, 0) AS total_alunos,
    COALESCE(pf.total_professores, 0) AS total_professores,
    e.endereco AS cidade,
    NULL::text AS estado
   FROM escolas e
     LEFT JOIN ( SELECT alunos.escola_id,
            count(*)::integer AS total_alunos
           FROM alunos
          GROUP BY alunos.escola_id) a ON a.escola_id = e.id
     LEFT JOIN ( SELECT p.escola_id,
            count(*)::integer AS total_professores
           FROM profiles p
          WHERE p.role = 'professor'::user_role
          GROUP BY p.escola_id) pf ON pf.escola_id = e.id;
public|vw_admin_activity_feed_enriched|false| SELECT e.id,
    e.escola_id,
    e.occurred_at,
    e.event_family,
    e.event_type,
    COALESCE(NULLIF(p.nome, ''::text), NULLIF(au.raw_user_meta_data ->> 'full_name'::text, ''::text), 'Sistema'::text) AS actor_name,
    COALESCE(NULLIF(e.payload ->> 'headline'::text, ''::text),
        CASE
            WHEN e.event_type = 'PAGAMENTO_REGISTRADO'::text THEN 'Pagamento confirmado'::text
            WHEN e.event_type = 'NOTA_LANCADA_BATCH'::text THEN 'Lançamento de notas concluído'::text
            WHEN e.event_type = 'DOCUMENTO_EMITIDO'::text THEN 'Documento emitido'::text
            WHEN e.event_type ~~ 'MATRICULA_%'::text THEN 'Actualização de matrícula'::text
            WHEN e.event_type ~~ 'ADMISSAO_%'::text THEN 'Actualização de admissão'::text
            ELSE initcap(replace(lower(e.event_type), '_'::text, ' '::text))
        END) AS headline,
    COALESCE(NULLIF(e.payload ->> 'subline'::text, ''::text), e.payload ->> 'turma_nome'::text, e.payload ->> 'aluno_nome'::text, e.payload ->> 'documento_nome'::text) AS subline,
    NULLIF(e.payload ->> 'valor_numeric'::text, ''::text)::numeric(14,2) AS amount_kz,
    NULLIF(COALESCE(e.payload ->> 'turma_nome'::text, t.nome), ''::text) AS turma_nome,
    NULLIF(COALESCE(e.payload ->> 'aluno_nome'::text, a.nome_completo, a.nome), ''::text) AS aluno_nome,
    e.payload
   FROM admin_activity_events e
     LEFT JOIN profiles p ON p.user_id = e.actor_id
     LEFT JOIN auth.users au ON au.id = e.actor_id
     LEFT JOIN alunos a ON a.id::text = COALESCE(e.payload ->> 'aluno_id'::text, e.entity_id)
     LEFT JOIN turmas t ON t.id::text = COALESCE(e.payload ->> 'turma_id'::text, e.payload #>> '{turma,id}'::text[]);

public|vw_rotinas_compat|false| SELECT q.id,
    q.turma_id,
    NULL::uuid AS secao_id,

  WHERE COALESCE(hs.is_intervalo, false) = false AND hv.status = 'publicada'::text;
